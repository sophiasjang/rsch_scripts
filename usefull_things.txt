\4   \1   \2   \3

# remove all extremities in BCL residue
.* (C[0-9]{1,2}|C[AB][BC]|CM[ABCD]|C[GE]D|C[ABG]A|O[B12]D|OBB|O[12]A) *[ A-Z]BCL.*\n

# match whole line not containing word
^(.(?!word1|word2))*$
